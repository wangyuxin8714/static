     删除类方法的封装(removeClass)
         var box=$('box');
         function removeClass(node,cla){
	             通过空格转数组
	             var arr=node.className.split(' ');
	             为了不影响下标值的改变 所以倒序遍历arr
	             for(var i=arr.length-1;i>=0;i--){
		               if(arr[i]==cla){
			                  arr.splice(i,1);
			               }
	                         }
	             node.className=arr.join(" ");
		          }
         removeClass(box,'w');

     封装getElementsByTagName()方法

	 var node=document.body;
	 function byTagName(node,val){
		     var arr=[];
		     for(var i=0; i<node.children.length;i++){
			       if(node.children[i].nodeName==val.toUpperCase()){
				                             arr.push(node.children[i]);
				                                      }
				 			     }
		return arr;
						   }
